Trap frame of kernel interrupt

§Trap handler

You need to define a handler function like this:

use trapframe::TrapFrame;

#[no_mangle]
extern "sysv64" fn trap_handler(tf: &mut TrapFrame) {
    match tf.trap_num {
        3 => {
            println!("TRAP: BreakPoint");
            tf.rip += 1;
        }
        _ => panic!("TRAP: {:#x?}", tf),
    }
}
